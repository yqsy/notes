```bash
# 网卡

INTERFACE=enp3s0
cat << EOF | sudo tee /etc/network/interfaces
auto lo
iface lo inet loopback
auto br0
iface br0 inet dhcp
      bridge_ports ${INTERFACE}
      bridge_stp off
      bridge_maxwait 0
EOF

sudo systemctl restart networking.service

# 修改
/etc/default/ufw

DEFAULT_FORWARD_POLICY="ACCEPT"

sudo systemctl restart ufw.service
```
